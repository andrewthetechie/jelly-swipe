       ```python
       def rate_limit(max_requests: int, per_minutes: int = 1):
           """Decorator to enforce per-IP rate limiting on a Flask route.
           
           Args:
               max_requests: Maximum requests allowed per per_minutes window
               per_minutes: Time window in minutes (default 1)
           
           Returns 429 with Retry-After header when limit exceeded.
           Logs violations at WARNING level (D-18/D-19/D-20).
           """
           from functools import wraps
           def decorator(f):
               @wraps(f)
               def wrapped(*args, **kwargs):
                   ip = request.remote_addr  # D-12: ProxyFix populates this
                   endpoint = request.endpoint  # D-07: Flask endpoint name
                   allowed, retry_after = rate_limiter.check(
                       endpoint=endpoint,
                       ip=ip,
                       limit=max_requests,
                       per_minutes=per_minutes
                   )
                   if not allowed:
                       import math
                       retry_seconds = int(math.ceil(retry_after))
                       app.logger.warning(
                           "rate_limit_exceeded",
                           extra={
                               'request_id': get_request_id(),
                               'endpoint': endpoint,
                               'ip_address': ip,
                               'limit': max_requests,
                               'retry_after_seconds': retry_seconds,
                           }
                       )
                       resp_body, status = make_error_response(
                           "Rate limit exceeded",
                           429,
                           extra_fields={"retry_after": retry_seconds}
                       )
                       # Flask returns a tuple (response, status) from make_error_response
                       # We need a Response object to add headers
                       from flask import make_response as flask_make_response
                       response = flask_make_response(resp_body, status)
                       response.headers["Retry-After"] = str(retry_seconds)
                       return response
                   return f(*args, **kwargs)
               return wrapped
           return decorator
       ```
       
       Note: `make_error_response()` currently returns `(jsonify(body), status_code)` — a tuple. To add the `Retry-After` header, wrap with `flask.make_response()` to convert the tuple to a Response object, then set the header.
