```python
def generate():
    last_genre = None
    last_ready = None
    last_match_ts = None
    POLL = 1.5
    TIMEOUT = 3600

    # Per DB-02: Hold one persistent connection for the entire
    # stream lifetime instead of opening/closing per poll cycle.
    # WAL mode (set in init_db) eliminates file-lock contention
    # so concurrent readers don't block this connection.
    conn = sqlite3.connect(jellyswipe.db.DB_PATH)
    conn.row_factory = sqlite3.Row
    try:
        deadline = time.time() + TIMEOUT
        while time.time() < deadline:
            try:
                row = conn.execute(
                    'SELECT ready, current_genre, solo_mode, last_match_data FROM rooms WHERE pairing_code = ?',
                    (code,)
                ).fetchone()

                if row is None:
                    yield f"data: {json.dumps({'closed': True})}\n\n"
                    return

                ready = bool(row['ready'])
                genre = row['current_genre']
                solo = bool(row['solo_mode'])
                last_match = json.loads(row['last_match_data']) if row['last_match_data'] else None
                match_ts = last_match['ts'] if last_match else None

                payload = {}
                if ready != last_ready:
                    payload['ready'] = ready
                    payload['solo'] = solo
                    last_ready = ready
                if genre != last_genre:
                    payload['genre'] = genre
                    last_genre = genre
                if match_ts and match_ts != last_match_ts:
                    payload['last_match'] = last_match
                    last_match_ts = match_ts

                if payload:
                    yield f"data: {json.dumps(payload)}\n\n"

                time.sleep(POLL)
            except GeneratorExit:
                return
            except Exception:
                time.sleep(POLL)
    finally:
        conn.close()
```

**Test 2 — test_stream_heartbeat_on_idle (D-11):**

```python
def test_stream_heartbeat_on_idle(client, monkeypatch):
    """SSE stream sends : ping heartbeat when no data event for ~15 seconds."""
    _seed_stream_room("HB1", ready=0, current_genre="All")
    _set_session_room(client, "HB1")

    # Create time mock that advances 16 seconds between polls to trigger heartbeat
    call_count = 0
    real_start = time.time()

    def _time_with_gap():
        nonlocal call_count
        call_count += 1
        # First call: deadline calculation (real time)
        # Second call: first poll (real time)
        # Third call: after sleep — advance 16 seconds to trigger heartbeat
        # Fourth call: heartbeat sent, check deadline — advance past to exit
        if call_count <= 2:
            return real_start
        elif call_count == 3:
            return real_start + 16  # 16+ seconds gap triggers heartbeat
        else:
            return real_start + 3700  # past deadline, exit loop

    monkeypatch.setattr(time, "sleep", lambda _: None)
    monkeypatch.setattr(time, "time", _time_with_gap)

    response = client.get("/room/stream")
    data = response.data.decode()

    assert ": ping\n\n" in data, f"Expected heartbeat ': ping\\\\n\\\\n' in SSE stream, got: {repr(data)}"
```
